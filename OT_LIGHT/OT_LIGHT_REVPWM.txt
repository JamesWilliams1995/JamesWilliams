boolean Touch_getXY(void) {
  TSPoint p = ts.getPoint();
  pinMode(YP, OUTPUT);      // restore shared pins
  pinMode(XM, OUTPUT);
  digitalWrite(YP, HIGH);   // because TFT control pins
  digitalWrite(XM, HIGH);
  register volatile bool pressed = (p.z > ts.pressureThreshhold);

  if (pressed) {
    pixel_x = map(p.x, TS_LEFT, TS_RT, 0, 480);
    pixel_y = map(p.y, TS_TOP, TS_BOT, 0, 320);

    Serial.print(String("pixel_x  ") + pixel_x);
    Serial.println(String("pixel_y ") + pixel_y);

  }
  else {
    pixel_x = -1;
    pixel_y = -1;
  }

  return pressed;
}

double mapping (double x, double from_min, double from_max, double to_min, double to_max)
{
  return (x - from_min) * (from_max - from_min) / (to_max - to_min) + to_min;
}

  Serial.begin(9600);
  uint16_t ID = tft.readID();
  //Serial.print("TFT ID = 0x");
  //Serial.println(ID, HEX);
  if (ID == 0xD3D3) ID = 0x9486; // write-only shield
  tft.begin(ID);
  tft.setRotation(0);

  tft.fillRect(0, 0, 320, 480, BLACK);
  tft.setRotation(1);
  //tft.fillScreen(BLACK);
  tft.drawBitmap(50, 80 , epd_bitmap_CISTRON_LOGO__PNG___2_, 450, 150,  tft.color565(255, 0, 0));
  delay(1000);
  // PORTRAIT
  tft.setRotation(0);
  tft.fillScreen(WHITE);
  tft.setRotation(1);
  setupDisplay();

}

void loop(void) {

  Touch_getXY();

  MapPwm = (float)(PercentageTypecase / total);
  //Serial.println(String("MapPwm  ")+MapPwm);
  double RevMap = mapping (MapPwm, 0.00, 1.00, 1.00, 0.00);

  // Serial.println(String("value  ")+value);
  Pwm = static_cast<int>(RevMap * 255);

  Serial.println(String("Pwm  ") + Pwm);

  MapPwm1 = (float)(PercentageTypecase1 / total);
  double RevMap1 = mapping (MapPwm1, 0.00, 1.00, 1.00, 0.00);
  Pwm1 = static_cast<int>(RevMap1 * 255);

  if (page == 1) {
    analogWrite(ledPin, Pwm);
    analogWrite(ledPin_yellow, Pwm1);
    updatePage1();
  }
  else if (page == 2) {
    updatePage2();
  }
}
